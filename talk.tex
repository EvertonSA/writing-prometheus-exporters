\begin{frame}[fragile]
\frametitle{Side-car Counter}

\begin{lstlisting}
class SidecarCounter(metrics.MetricWrapperBase):
    _type = "counter"
    def _metric_init(self):
        self._value = values.ValueClass(
            self._type,
            self._name,
            self._name + "_total",
            self._labelnames,
            self._labelvalues,
        )
        self._created = values.ValueClass(
            self._type,
            self._name,
            self._name + "_created",
            self._labelnames,
            self._labelvalues,
        )
 \end{lstlisting}
 \end{frame}
 

\begin{frame}[fragile]
\frametitle{Set/Get Counter}
\begin{lstlisting}
    def set(self, what, *, when):
        self._value.set(what)
        self._created.set(when)
    def _child_samples(self):
        return (
            ("_total", {}, self._value.get()),
            ("_created", {}, self._created.get()),
        )
\end{lstlisting}
\end{frame}
